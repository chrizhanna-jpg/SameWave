/** Must match Clerk Dashboard → Native applications → “Allowlist for mobile SSO redirect”. */
function getGoogleSsoRedirectUrl(): string {
  return AuthSession.makeRedirectUri(
    Platform.OS === "android"
      ? { scheme: Constants.expoConfig?.android?.package ?? "app.echo.samesame", path: "callback" }
      : Platform.OS === "ios"
        ? { scheme: Constants.expoConfig?.ios?.bundleIdentifier ?? "app.echo.samesame", path: "callback" }
        : { scheme: "same-same" },
  );
}
